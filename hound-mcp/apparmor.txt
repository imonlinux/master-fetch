#include <tunables/global>

profile hound_mcp flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>
  #include <abstractions/python>
  #include <abstractions/nameservice>

  # Capabilities for network and browser operations
  capability net_raw,
  capability setuid,
  capability setgid,

  # Network access
  network inet tcp,
  network inet stream,

  # Allow access to application directory
  /app/** r,
  /app/src/** rm,

  # Allow access to data directory
  /data/** rw,
  /data/.hound/** rwkl,

  # Allow tmpfs and shared memory (for Chromium)
  /tmp/** rw,
  /dev/shm/** rw,
  /dev/shm/** rwkl,

  # Python and Playwright requirements
  /usr/bin/** rix,
  /usr/lib/** r,
  /usr/local/lib/** r,
  /lib/** r,
  /lib64/** r,

  # Allow read access to system libraries
  /etc/** r,
  /proc/** r,

  # Playwright browser files
  /ms-playwright/** rw,
  /root/.cache/ms-playwright/** rw,

  # Deny sensitive areas (these are default denies, explicitly stated)
  deny /root/.ssh/** rw,
  deny /home/**/.ssh/** rw,
}
